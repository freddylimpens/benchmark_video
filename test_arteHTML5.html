Demo page : force HTML5

//mandatory on non arte.tv domains
document.domain = "arte.tv";
$(window).load(function() {
    //listen to the arte_vp_player_config_ready event
    $('.video-container').on('arte_vp_player_config_ready',function() {
        //force HTML5
        $(this).find('iframe')[0].contentWindow.arte_vp.parameters.config.primary = "html5";
    });
});
